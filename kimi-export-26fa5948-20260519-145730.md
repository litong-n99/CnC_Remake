```typescript
export class UnitCollision {
  static locomotor: Locomotor | null = null;

  static isPositionBlocked(x: number, y: number, excludeId: string): boolean {
    const cx = Math.round(x), cy = Math.round(y);
    const self = GameObjectManager.getInstance().get(excludeId);
    if (self) {
      const mobile = (self as unknown as { logic?: { mobile?: { getFromCell:()=>{x:number;y:number} } } }).logic?.mobile;
      if (mobile) {
        const from = mobile.getFromCell();
        if (from.x === cx && from.y === cy) return false; // 仅放行 fromCell
      }
    }
    if (this.locomotor && self) {
      return !this.locomotor.canMoveFreelyInto(self, cx, cy, 3 /* BlockedByActor.All */);
    }
    return this.isPositionBlockedLegacy(x, y, excludeId);
  }

  static getBlockedCells(excludeId: string): Set<string> {
    const blocked = new Set<string>();
    for (const obj of GameObjectManager.getInstance().getAll()) {
      if (!obj.isAlive() || obj.id === excludeId || obj.type !== GameObjectType.Unit) continue;
      const mobile = (obj as unknown as { logic?: { mobile?: { getFromCell:()=>{x:number;y:number}; getToCell:()=>{x:number;y:number} } } }).logic?.mobile;
      if (mobile) {
        const from = mobile.getFromCell(), to = mobile.getToCell();
        blocked.add(`${from.x},${from.y}`);
        if (to.x !== from.x || to.y !== from.y) blocked.add(`${to.x},${to.y}`);
      } else {
        blocked.add(`${Math.round(obj.x)},${Math.round(obj.y)}`);
      }
    }
    return blocked;
  }
}
```

```typescript
export class Mobile {
  private fromCell: { x: number; y: number };
  private toCell: { x: number; y: number };
  private isDualOccupying = false;
  isBlocking = false;

  enterNewCell(node: PathNode): void {
    if (node.x === this.toCell.x && node.y === this.toCell.y) return;
    const currentCx = Math.round(this.gameObject.x);
    const currentCy = Math.round(this.gameObject.y);
    if (this.toCell.x === currentCx && this.toCell.y === currentCy) {
      this.fromCell = { ...this.toCell };
    } else {
      this.fromCell = { x: currentCx, y: currentCy };
    }
    this.toCell = { x: node.x, y: node.y };
    this.isDualOccupying = true;
    this.updateOccupancy();
  }

  onStop(): void {
    if (!this.isDualOccupying) return;
    this.fromCell = { ...this.toCell };
    this.isDualOccupying = false;
    this.updateOccupancy();
  }

  cellIsEvacuating(targetCx: number, targetCy: number, excludeId: string): boolean {
    const actors = this.actorMap.getActorsAt(targetCx, targetCy);
    for (const actor of actors) {
      if (actor.id === excludeId) continue;
      const mobile = (actor as unknown as { logic?: { mobile?: { isMovingBetweenCells:()=>boolean; getFromCell:()=>{x:number;y:number} } } }).logic?.mobile;
      if (!mobile || !mobile.isMovingBetweenCells()) return false;
      const from = mobile.getFromCell();
      if (from.x !== targetCx || from.y !== targetCy) return false;
    }
    return true;
  }
  // ...getFromCell, getToCell, getAdjacentCell, updateOccupancy...
}
```

**Summary:**
移动控制器 — A* 路径插值 + OpenRA 风格 6 步阻塞 fallback 链。

**Key elements:**
- `moveTo()`：重置所有阻塞状态（`hasWaited=false`, `waitRemainingMs=0`, `repathAttempts=0`），初始化路径，触发 `onPathNodeReached`
- `update()`：每帧循环步进；到达节点触发 `onPathNodeReached`；阻塞时进入 `handleBlocked`
- `handleBlocked()`：Notify→Wait→CheckEvac→Repath（**使用 `Math.round(controller.x)` 起点**）→BackUp→GiveUp
- `stop()`：清理状态并回调 `onStop`

**Latest version:**
```typescript
export class UnitMovement {
  onPathNodeReached: ((node: PathNode) => void) | null = null;
  onStop: (() => void) | null = null;
  private hasWaited = false;
  private waitRemainingMs = 0;
  private repathAttempts = 0;
  private static readonly WAIT_DURATION_MS = 650;
  private static readonly MAX_REPATH_ATTEMPTS = 3;

  moveTo(controller: UnitController, targetX: number, targetY: number, pathfinder: Pathfinder): boolean {
    const blockedCells = UnitCollision.getBlockedCells(controller.unitId);
    const path = pathfinder.findPath(
      Math.round(controller.x), Math.round(controller.y),
      targetX, targetY, blockedCells
    );
    if (!path || path.length <= 1) return false;
    this.path = path;
    this.pathIndex = 1;
    this.isMoving = true;
    this.pathfinder = pathfinder;
    this.hasWaited = false;
    this.waitRemainingMs = 0;
    this.repathAttempts = 0;
    controller.stateMachine.transition(UnitState.Moving);
    controller.isDriving = true;
    controller.moveTarget = { x: targetX, y: targetY };
    if (this.path.length > 1) this.onPathNodeReached?.(this.path[1]);
    return true;
  }

  private handleBlocked(controller: UnitController, deltaTime: number, targetX: number, targetY: number): void {
    const targetCx = Math.round(targetX), targetCy = Math.round(targetY);
    controller.mobile?.notifyBlockersAt(targetCx, targetCy, controller.unitId, this.pathfinder!, UnitCollision.locomotor);
    // Step 2: Wait
    if (!this.hasWaited) { this.hasWaited = true; this.waitRemainingMs = UnitMovement.WAIT_DURATION_MS; }
    this.waitRemainingMs -= deltaTime;
    if (this.waitRemainingMs > 0) { /* face target & return */ return; }
    // Step 3: Check evacuating
    if (controller.mobile?.cellIsEvacuating(targetCx, targetCy, controller.unitId)) {
      this.waitRemainingMs = UnitMovement.WAIT_DURATION_MS * 0.5; return;
    }
    // Step 4: Repath — 使用实际位置作为起点
    if (this.repathAttempts < UnitMovement.MAX_REPATH_ATTEMPTS) {
      this.repathAttempts++;
      const blockedCells = UnitCollision.getBlockedCells(controller.unitId);
      const startCell = { x: Math.round(controller.x), y: Math.round(controller.y) };
      const newPath = this.pathfinder?.findPath(startCell.x, startCell.y, targetCx, targetCy, blockedCells);
      if (newPath && newPath.length > 1) {
        this.path = newPath; this.pathIndex = 1; this.hasWaited = false; this.waitRemainingMs = 0; this.repathAttempts = 0;
        this.onPathNodeReached?.(this.path[1]);
        return;
      }
      this.waitRemainingMs = UnitMovement.WAIT_DURATION_MS * 0.5; return;
    }
    // Step 5: Back up
    if (controller.mobile?.isBlocking) {
      const backupCell = controller.mobile.getAdjacentCell({ x: targetCx, y: targetCy }, UnitCollision.locomotor);
      if (backupCell) {
        this.path = [backupCell]; this.pathIndex = 0;
        this.hasWaited = false; this.waitRemainingMs = 0; this.repathAttempts = 0;
        controller.mobile.isBlocking = false;
        controller.mobile.enterNewCell(backupCell);
        return;
      }
    }
    // Step 6: Give up
    this.stop(controller);
  }

  private stop(controller: UnitController): void {
    this.isMoving = false; this.pathfinder = null;
    this.hasWaited = false; this.waitRemainingMs = 0; this.repathAttempts = 0;
    controller.stateMachine.transition(UnitState.Idle);
    controller.isDriving = false; controller.moveTarget = undefined;
    this.onStop?.();
  }
}
```

### 新 Phase 5.5 核心原则

**新方案核心原则**：

3. **A* 负责宏观绕开**：初始寻路时排除其他单位位置，路径本身就不会主动穿过其他单位
4. **阻塞时自驱 fallback**：暂停→重寻路（到原始目标）→Nudge→GiveUp，不依赖其他单位配合
5. **增量可验证**：每个子任务完成后都能独立测试，不依赖后续任务

```typescript
  move(id: string, fromX: number, fromY: number, toX: number, toY: number): void

}
```
